# SSH Connection Test Commands
# Copy and paste these commands one by one to test your server connection

# 1. Test default SSH port (22) - Replace YOUR_SERVER_IP with your actual server IP
ssh -i ~/.ssh/id_ed25519_4blocksdevs -p 22 ubuntu@YOUR_SERVER_IP

# 2. Test if server is reachable on port 22
Test-NetConnection -ComputerName YOUR_SERVER_IP -Port 22

# 3. Test different common SSH ports
Test-NetConnection -ComputerName YOUR_SERVER_IP -Port 2222
Test-NetConnection -ComputerName YOUR_SERVER_IP -Port 2200

# 4. Check which user accounts exist on the server
ssh -i ~/.ssh/id_ed25519_4blocksdevs ubuntu@YOUR_SERVER_IP "whoami"
ssh -i ~/.ssh/id_ed25519_4blocksdevs root@YOUR_SERVER_IP "whoami"

# 5. Get server SSH configuration
ssh -i ~/.ssh/id_ed25519_4blocksdevs ubuntu@YOUR_SERVER_IP "sudo cat /etc/ssh/sshd_config | grep Port"

# REPLACE YOUR_SERVER_IP with your actual server IP address!
# Example: 192.168.1.100 or yourserver.com
